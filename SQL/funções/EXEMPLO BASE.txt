CREATE OR REPLACE FUNCTION FN_DEV_DESCR(PCOD PRODUTO.COD_PRODUTO%TYPE)
RETURN VARCHAR2
AS
	VDESCR PRODUTO.DESCRICAO%TYPE;
BEGIN
	SELECT DESCRICAO INTO VDESCR
	FROM PRODUTO
	WHERE COD_PRODUTO = PCOD;
	REUTNR VDESCR;
END;

	     FROM ITEM_PEDIDO;
